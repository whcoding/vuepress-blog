    Mysql binlog

    # mysql binlog

    # binlog 常用命令

    1. 查看是否开启
    show variables like 'log_%';
    1
    1. 查看当前使用的biglog 日志
    show master status;
    1
    1. 查看全部biglog日志
    show master LOGS;
    1
    1. 刷新 binlog 日志. 产生一个新的文件
    flush logs;
    1
    1. 重置(清空)所有 binlog 日志
    reset master;
    1

    # 开启binlog

    • 在my.cnf中加入
    [mysqld]
server_id = 1
log_bin = /var/log/mysql/mysql-bin.log
expire_logs_days = 1
max_binlog_size  = 100M
